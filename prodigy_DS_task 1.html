# Extract the 'Age' column, dropping NaN values
for i in df.columns:
    mn = df[i].isna().sum()
    if mn < 300 and df[i].dtype == 'object':
        df[i] = df[i].fillna(df[i].mode()[0])
    elif mn < 300 and df[i].dtype != 'object':
        df[i] = df[i].fillna(df[i].mean())
    else:
        df.drop(columns=i, inplace=True)
